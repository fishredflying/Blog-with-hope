    def format_speed(self, bytes_per_sec):
        units = ['B/s', 'KB/s', 'MB/s']
        unit_idx = 0
        while bytes_per_sec >= 1024 and unit_idx < 2:
            bytes_per_sec /= 1024
            unit_idx += 1
        return f"{bytes_per_sec:.2f} {units[unit_idx]}"
    
    def download_chunk(self, url, start, end, part_num):
        self.log("çº¿ç¨‹-%då·²åˆ†é…%d-%då­—èŠ‚ï¼Œæ­£åœ¨ä¸‹è½½ã€‚" % (part_num,start,end))
        try:
            headers = {'Range': f'bytes={start}-{end}'}
            response = requests.get(url, headers=headers, stream=True)
            temp_file = self.temp_dir / f"part_{part_num}"
            now_id=start
            with open(temp_file, 'wb') as f:
                for chunk in response.iter_content(chunk_size=self.chunk_size) :
                    if chunk :
                        f.write(chunk)
                        self.download_queue.put(len(chunk))
                        self.downloaded_total+=len(chunk)
                        now_id+=len(chunk)
                    self.root.update()
            self.log("çº¿ç¨‹-%då·²å®Œæˆï¼" % part_num)
        except Exception as e:
            self.log(f"çº¿ç¨‹{part_num}ä¸‹è½½å¤±è´¥: {str(e)}")
            self.download_queue.put(0)  # ç¡®ä¿ç›‘æ§ç»§ç»­

                self.log("çº¿ç¨‹-%då·²é‡è¯•ï¼" % part_num)

    def monitor_progress(self, threads, save_path):
        def update():
            # æ›´æ–°é€Ÿåº¦æ˜¾ç¤º
            now = time.time()
            time_diff = now - self.last_update
            if time_diff >= 1:
                self.speed = self.downloaded_bytes / time_diff
                self.speed_label.config(text=f"é€Ÿåº¦: {self.format_speed(self.speed)}")
                self.last_update = now
                self.downloaded_bytes = 0
            
            # å¤„ç†ä¸‹è½½æ•°æ®
            while not self.download_queue.empty():
                chunk_size = self.download_queue.get()
                self.progress += chunk_size
                self.downloaded_bytes += chunk_size
                self.progress_bar['value'] = self.progress
            
            # æ£€æŸ¥çº¿ç¨‹çŠ¶æ€
            if any(t.is_alive() for t in threads):
                self.root.after(100, update)
            else:
                self.nwin.destroy()

                finally:
                    self.is_downloading = False

                    self.root.update()
        
        self.root.after(100, update)
